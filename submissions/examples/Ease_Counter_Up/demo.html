ease-count-up.css — the trick 
/* 1 — register as animatable integer */
@property --ease-count {
  syntax: "<integer>";
  inherits: false;
  initial-value: 0;
}

/* 2 — animate 0 → target */
@keyframes ease-count-up {
  from { --ease-count: 0 }
  to   { --ease-count: var(--ease-count-target) }
}

/* 3 — wire to CSS counter */
.count-up {
  counter-reset: n var(--ease-count);
}
.count-up::after {
  content: counter(n);
}

/* 4 — trigger by adding .counting */
.count-up.counting {
  animation: ease-count-up 2.4s cubic-bezier(0.16,1,0.3,1) both;
}

/* 5 — usage */
<span class="count-up elastic"
      style="--ease-count-target: 4200"></span>
